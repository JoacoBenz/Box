flowchart TD
    %% ── INICIO ──
    START(("🟢 Inicio")) --> CREATE

    %% ── SOLICITANTE ──
    subgraph SOL["👤 SOLICITANTE"]
        CREATE["📝 Crear Solicitud\n(borrador)"]
        EDIT["✏️ Editar items,\njustificación, urgencia"]
        SEND{"Enviar\nsolicitud?"}
        CREATE --> EDIT
        EDIT --> SEND
    end

    SEND -->|Sí| ENVIADA["📨 ENVIADA"]
    SEND -->|No| EDIT

    %% ── AUTO-VALIDACIÓN ──
    ENVIADA --> CHECK_AUTO{"¿Solicitante es\nResponsable de Área?"}
    CHECK_AUTO -->|Sí| AUTO_VAL["⚡ Auto-validada"]
    AUTO_VAL --> VALIDADA
    CHECK_AUTO -->|No| RESP_REV

    %% ── RESPONSABLE DE ÁREA ──
    subgraph RESP["👥 RESPONSABLE DE ÁREA"]
        RESP_REV{"Revisar\nsolicitud"}
        RESP_VAL["✅ Validar"]
        RESP_DEV["↩️ Devolver\n(observaciones)"]
    end

    RESP_REV -->|Aprobar| RESP_VAL
    RESP_REV -->|Observaciones| RESP_DEV
    RESP_VAL --> VALIDADA["✅ VALIDADA"]
    RESP_DEV --> DEV_RESP["🔙 DEVUELTA_RESP"]
    DEV_RESP -->|"Solicitante corrige\ny re-envía"| ENVIADA

    %% ── DIRECTOR ──
    subgraph DIR["👔 DIRECTOR"]
        DIR_REV{"Revisar\nsolicitud"}
        DIR_APR["✅ Aprobar"]
        DIR_DEV["↩️ Devolver\n(observaciones)"]
        DIR_RECH["❌ Rechazar\n(motivo)"]
    end

    VALIDADA --> DIR_REV
    DIR_REV -->|Aprobar| DIR_APR
    DIR_REV -->|Observaciones| DIR_DEV
    DIR_REV -->|Rechazar| DIR_RECH

    DIR_APR --> EN_COMPRAS["🛒 EN_COMPRAS"]
    DIR_DEV --> DEV_DIR["🔙 DEVUELTA_DIR"]
    DIR_RECH --> RECHAZADA["🚫 RECHAZADA"]

    DEV_DIR -->|"Solicitante corrige\ny re-envía"| ENVIADA
    DEV_DIR -->|"Responsable\nre-valida"| VALIDADA

    %% ── COMPRAS ──
    subgraph COMP["🛒 COMPRAS"]
        COMP_PROC{"Procesar\nsolicitud"}
        COMP_PRIO["📋 Asignar prioridad\ny día de pago"]
    end

    EN_COMPRAS --> COMP_PROC
    COMP_PROC --> COMP_PRIO
    COMP_PRIO --> PAGO_PROG["💳 PAGO_PROGRAMADO"]

    %% ── TESORERÍA ──
    subgraph TES["💰 TESORERÍA"]
        TES_COMPRA{"Registrar compra"}
        TES_REG["📄 Cargar comprobante,\nmonto, proveedor,\nmedio de pago"]
    end

    PAGO_PROG --> TES_COMPRA
    TES_COMPRA --> TES_REG
    TES_REG --> ABONADA["💰 ABONADA"]

    %% ── RECEPCIÓN (SOLICITANTE) ──
    subgraph RECEP["📦 RECEPCIÓN"]
        RECEP_CHECK{"¿Recepción\nconforme?"}
        RECEP_OK["✅ Conforme"]
        RECEP_OBS["⚠️ Con observaciones\n(tipo problema + detalle)"]
    end

    ABONADA --> RECEP_CHECK
    RECEP_CHECK -->|Sí| RECEP_OK
    RECEP_CHECK -->|No| RECEP_OBS

    RECEP_OK --> CERRADA["🏁 CERRADA"]
    RECEP_OBS --> RECIB_OBS["⚠️ RECIBIDA_CON_OBS"]

    %% ── CIERRE ──
    RECIB_OBS -->|"Compras/Tesorería\nresuelve y cierra"| CERRADA

    %% ── ANULACIÓN (cualquier momento) ──
    ANULADA["🚫 ANULADA"]
    ENVIADA -.->|"Anular\n(solicitante/director/admin)"| ANULADA
    VALIDADA -.->|Anular| ANULADA
    EN_COMPRAS -.->|Anular| ANULADA
    PAGO_PROG -.->|Anular| ANULADA

    %% ── FIN ──
    CERRADA --> FIN(("🔴 Fin"))
    RECHAZADA --> FIN
    ANULADA --> FIN

    %% ── ESTILOS ──
    classDef solicitante fill:#dbeafe,stroke:#3b82f6,color:#1e293b

    class CREATE,EDIT,SEND solicitante
    class RESP_REV,RESP_VAL,RESP_DEV responsable
    class DIR_REV,DIR_APR,DIR_DEV,DIR_RECH director

    class TES_COMPRA,TES_REG tesoreria
    class RECEP_CHECK,RECEP_OK,RECEP_OBS solicitante
